\section*{Project Description}
Classic Tetris implementation in C using ncurses library for school21

\section*{Game Rules}
\begin{itemize}
\item Blocks fall from the top
\item Player moves and rotates blocks
\item Goal: complete horizontal lines
\item Completed lines disappear, player gets points:
1 line - 100 \\
2 lines - 300 \\
3 lines - 800 \\
4 lines - 1500
\item After every 600 points, the game level increases, which increases the speed of falling figures. The maximum level is -10
\item Game ends when blocks reach the top
\end{itemize}

\section*{Controls}
\begin{itemize}
\item Left: left arrow
\item Right: right arrow
\item Down: down arrow (speed up)
\item Rotate: up arrow
\item Pause: P
\end{itemize}

\section*{Build and Run}
\begin{verbatim}
make install
./build/tetris.o
\end{verbatim}
